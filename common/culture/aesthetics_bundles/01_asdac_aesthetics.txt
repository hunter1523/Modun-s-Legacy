aesthetics_ukrainian = {
	is_shown = {
		scope:character = {
			culture = {
				has_cultural_pillar = heritage_east_slavic
			}
			any_sub_realm_county = {
				any_this_title_or_de_jure_above = {
					OR = {
						this = title:k_ruthenia
						this = title:k_galicia-volhynia
						this = title:k_hungary
						this = title:k_moldavia
						this = title:k_pontic_steppe
						this = title:k_caucasus
					}
				}
			}
			NOT = {
				culture = culture:ukrainian
			}
		}
	}

	name_list = name_list_ukrainian

	coa_gfx = { polish_coa_gfx west_slavic_group_coa_gfx western_coa_gfx }
	building_gfx = { western_building_gfx }
	clothing_gfx = { northern_clothing_gfx }
	unit_gfx = { eastern_unit_gfx }
}
