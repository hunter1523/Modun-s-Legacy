is_china_government = {
	scope = character
	is_valid = {
		any_player = {
			root = THIS
		}
		AND = {
			has_title = title:e_celestia_china
			has_government = chinese_government
		}
	}
}
is_china_government_or_vassal = {
	scope = character
	is_valid = {
		any_player = {
			root = THIS
		}
		OR = {
			AND = {
				has_title = title:e_celestia_china
				has_government = chinese_government
			}
			top_liege = {
				AND = {
					has_title = title:e_celestia_china
					has_government = chinese_government
				}
			}
		}
	}
}
