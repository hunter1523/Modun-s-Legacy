
on_game_start = {
	on_actions = {
		name_nuosu_starting_characters
	}
}
on_birth_mother = {
	on_actions = {
		name_nuosu_characters_on_birth
	}
}

name_nuosu_starting_characters = {
	effect = {
		every_living_character = {
			limit = {
				exists = father
				is_male = yes
				culture = { has_name_list = name_list_yi }
			}
			name_yi_character_with_fathers_name_effect = yes
		}
	}
}

name_nuosu_characters_on_birth = {
	trigger = {
		scope:child = {
			culture = { has_name_list = name_list_yi }
		}
		scope:father = {
			culture = { has_name_list = name_list_yi }
		}
	}
	effect = {
		scope:child = {
			assign_random_yi_name_effect = yes
			name_yi_character_with_fathers_name_effect = yes
		}
	}
}
