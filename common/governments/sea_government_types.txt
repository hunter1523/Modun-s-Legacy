		scope:vassal = {
			if = {
				limit = {

				if = {

						is_powerful_vassal = yes
					}
					value = -15
				}
			}
		}
	}
	opinion_of_liege_desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					scope:vassal = {
						NOT = {

						}
						is_powerful_vassal = yes
					}
				}
				desc = "GOVERNMENT_MANDALA_NOT_ALLIED_POWERFUL"
			}
			triggered_desc = {
				trigger = {
					scope:vassal = {
						NOT = {
							is_allied_to = scope:liege
						}
						is_powerful_vassal = no
					}
				}
				desc = "GOVERNMENT_MANDALA_NOT_ALLIED"
			}
		}
