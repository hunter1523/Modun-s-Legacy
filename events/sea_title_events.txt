# Adopt a new name when certain cultures gain a duke, king, or emperor title
sea_title_event.0001 = {
	type = character_event
	title = sea_title_event.0001.title
	desc = sea_title_event.0001.desc

	theme = crown
	left_portrait = root

    trigger = {
    	OR = {
    		has_government = mandala_government
    		has_government = feudal_government
    		has_government = clan_government
    	}
    	primary_title = scope:title
		highest_held_title_tier >= tier_duchy
    	NOT = { has_character_flag = has_regnal_name }
    	OR = {
    		culture = { has_name_list = name_list_chamic }
    		culture = { has_name_list = name_list_khmer }
    	}
    }

	immediate = {
		if = {
			limit = {
		    	OR = {
		    		culture = { has_name_list = name_list_chamic }
		    		culture = { has_name_list = name_list_khmer }
		    	}
			}

	        root = { get_chamic_names_effect = yes }
	        root = { get_chamic_names_effect = yes }
	        root = { get_chamic_names_effect = yes }
	        root = { get_chamic_names_effect = yes }
	        root = { get_chamic_names_effect = yes }

sea_title_event.2000 = {
	hidden = yes

	trigger = {
		has_government = anarchist_government
		scope:title.tier = tier_county
		domain_size > 1
	}
	immediate = {
		scope:title.holder = {
			trigger_event = {
				id = sea_title_event.2001
				days = { 21 35 }
			}
		}
	}
}

sea_title_event.2001 = {
	type = character_event
	title = sea_title_event.2001.t
	desc = sea_title_event.2001.desc
	theme = realm
	right_portrait = scope:new_count

	trigger = {
		NOT = { scope:title = root.capital_county }
		scope:title.holder = root
	}

	immediate = {
		# Create a new ruler of the same culture and faith of the county and make them your vassal
		create_character = {
			location = scope:title.title_province
			template = generic_peasant_character
			faith = scope:title.faith
			culture = scope:title.culture
			save_scope_as = new_count
		}
	}

	option = {
		# County goes to the new character
		name = sea_title_event.2001.a

		give_commune_county_effect = yes

		ai_chance = {
			base = 100
		}
	}
	option = {
		# Keep it for yourself. People will be mad.
		name = sea_title_event.2001.b

		add_character_modifier = {
			modifier = stateless_unpopular_modifier
			days = 7300
		}

		ai_chance = {
			base = 0
		}
	}
}

sea_title_event.2002 = {
	hidden = yes

	trigger = {
		has_government = anarchist_government
		scope:title = {
			can_title_have_law_general_trigger = yes
			NOT = { has_title_law = anarchist_elective_succession_law }
		}
	}
	immediate = {
		scope:title = {
			add_title_law = anarchist_elective_succession_law
		}
	}
}


sea_title_event.3000 = {
	hidden = yes

	trigger = {
		scope:title = { has_title_law = anarchist_elective_succession_law }
		scope:title.tier = tier_duchy
	}

	immediate = {
		if = {
			limit = {
				exists = scope:title.previous_holder
				dynasty = scope:title.previous_holder.dynasty
			}
			# Raise authority
			if = {
				limit = { has_realm_law = stateless_authority_0 }
				add_realm_law = stateless_authority_1
			}
			else_if = {
				limit = { has_realm_law = stateless_authority_1 }
				add_realm_law = stateless_authority_2
			}
			else_if = {
				limit = { has_realm_law = stateless_authority_2 }
				add_realm_law = stateless_authority_3
			}
		}
		else = {
			# Reset authority
			if = {
				limit = { NOT = { has_realm_law = stateless_authority_0 } }
				add_realm_law = stateless_authority_0
			}
		}
	}
}
