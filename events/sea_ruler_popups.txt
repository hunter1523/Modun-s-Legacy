# Fires the relevant character event if there is one
ruler_popups.0001 = {
	type = character_event
	hidden = yes

	immediate = {
		if = {
			limit = { this = character:421011 } # Youshilong
			trigger_event = {
				id = ruler_popups.0002
				days = 1
			}
		}
		if = {
			limit = { this = character:sailendra100 }
			trigger_event = {
				id = ruler_popups.0003
				days = 1
			}
		}
	}
}

			trigger_event = {
				id = ruler_popups.0004
				days = 4
			}
		}

			base = 10
		}
	}

	# Let us seek peace
	option = {
		name = ruler_popups.0003.b

		character:189042 = {
			trigger_event = {
				id = ruler_popups.0005
				days = 4
			}
		}

		ai_chance = {
			base = 100
		}
	}

	after = {
		show_as_tooltip = {

			}
		}
	}

ruler_popups.0005 = {
	type = character_event
	title = ruler_popups.0005.t
	desc = ruler_popups.0005.desc
	theme = war

	right_portrait = {
		character = character:sailendra100
		animation = beg
	}
	left_portrait = {
		character = character:189042
		animation = personality_cynical
	}

	# We will complete the conquest
	option = {
		name = ruler_popups.0005.a

		# Break tributary
		# Rivalry
		# Claim on palembang

		character:sailendra100 = {
			free_tributary_effect = yes
			add_truce_both_ways = {
				character = character:189042
				days = 365
				name = TRUCE_BROKEN_TRIBUTARY
			}
			set_relation_rival = {
				reason = rival_raided_me
				target = character:189042
			}
		}

		add_pressed_claim = title:d_palembang

		ai_chance = {
			base = 10
		}
	}

	# Know your place, fool
	option = {
		name = ruler_popups.0005.b
		custom_tooltip = ruler_popups.0005.b.tooltip

		# Maintain tributary

		ai_chance = {
			base = 20
		}
	}

	# Let us be friends
	option = {
		name = ruler_popups.0005.c

		add_opinion = {
			target = character:sailendra100
			modifier = respect_opinion
			opinion = 40
		}
		reverse_add_opinion = {
			target = character:sailendra100
			modifier = respect_opinion
			opinion = 40
		}

		# Break tributary

		character:sailendra100 = {
			free_tributary_effect = yes
			add_truce_both_ways = {
				character = character:189042
				days = 365
				name = TRUCE_BROKEN_TRIBUTARY
			}

			trigger_event = {
				id = ruler_popups.0006
				days = 4
			}
		}

		ai_chance = {
			base = 100
		}
	}
}

# Sailendra wishes to be reinstated as ruler of Kedah
ruler_popups.0007 = {
	type = character_event
	title = ruler_popups.0007.t
	desc = ruler_popups.0007.desc
	theme = war

	right_portrait = {
		character = character:sailendra100
		animation = worry
	}
	left_portrait = {
		character = character:189042
		animation = personality_compassionate
	}

	immediate = {
		title:c_kedah = {
			save_scope_as = kedah
		}
	}

	# Ok
	option = {
		name = ruler_popups.0007.a

		title:c_kedah = {
			save_scope_as = kedah
		}

		# Return Kedah

		create_title_and_vassal_change = {
			type = granted
			save_scope_as = title_change
			add_claim_on_loss = no
		}
		title:c_kedah = {
			change_title_holder = {
				holder = character:sailendra100
				change = scope:title_change
			}
		}
		resolve_title_and_vassal_change = scope:title_change

		reverse_add_opinion = {
			target = character:sailendra100
			modifier = kindness_opinion
			opinion = 20
		}

		ai_chance = {
			base = 100
		}
	}

	# Lol no
	option = {
		name = ruler_popups.0007.b

		# gain prestige
		add_prestige = 250

		reverse_add_opinion = {
			target = character:sailendra100
			modifier = refusal_opinion
			opinion = -40
		}

		ai_chance = {
			base = 20
		}
	}
}
